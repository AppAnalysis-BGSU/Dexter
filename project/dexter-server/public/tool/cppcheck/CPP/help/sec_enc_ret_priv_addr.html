Bad Code 
#include 
class Line
{
private:
    int foo;
public:
    Line()
    {
      foo = 1941;
    }
    int* getFooPtr()
    {
      return &foo;
    }
};


int main()
{
   Line a;

   int* p_foo2 = a.getFooPtr();
   *p_foo2 = 101; // break encapsulation
   std::cout << "foo=" << *a.getFooPtr() << std::endl;
   return 0;
}

#include 
class Line

    int foo;
public:
    Line()
    {
      foo = 1941;
    }
    int getFoo()
    {
      return foo;
    }
};


int main()
{
   Line a;

   int p_foo2 = a.getFoo();
   p_foo2 = 101; // can not break encapsulation
   std::cout << "foo=" << a.getFoo() << std::endl;
   return 0;
}
